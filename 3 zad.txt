<script>
	var broj =parseInt(prompt("Unesite broj:"));
	faktorijel(broj);
	function faktorijel(n, pre){
		if(pre ==null){
			pre = 0;
		}
		if(n == 0){
			return 1;
		} else {
			pre = faktorijel(n - 1, pre);
			document.write(n + "! = " + n * pre + "<br>");
			return n * pre;
		}
	}
